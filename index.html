// Kavanozları tanımla const jars = [ { x: 100, y: 150, size: 20, collected: false }, { x: 300, y: 200, size: 20, collected: false }, // Diğer kavanozları da buraya ekle ]; function drawJars() { jars.forEach(jar => { if (!jar.collected) { ctx.fillStyle = 'green'; ctx.fillRect(jar.x, jar.y, jar.size, jar.size); } }); } function checkCollisions() { jars.forEach(jar => { if (!jar.collected && playerX < jar.x + jar.size && playerX + player.offsetWidth > jar.x && playerY < jar.y + jar.size && playerY + player.offsetHeight > jar.y) { jar.collected = true; jarsCollected++; } }); } function draw() { ctx.clearRect(0, 0, canvas.width, canvas.height); drawJars(); // Diğer çizim ve karakter checkCollisions(); requestAnimationFrame(draw); } let jarsCollected = 0; draw(); let gameStarted = false; const startBtn = document.getElementById('startBtn'); startBtn.addEventListener('click', () => { gameStarted = true; startBtn.style.display = 'none'; // Butonu gizle }); document.addEventListener('keydown', (e) => { if (!gameStarted) return; // Oyun başlamadıysa hareket etme const step = 5; if (e.key === 'ArrowUp') playerY -= step; if (e.key === 'ArrowDown') playerY += step; if (e.key === 'ArrowLeft') playerX -= step; if (e.key === 'ArrowRight') playerX += step; // Sınırları kontrol et playerX = Math.max(0, Math.min(playerX, canvas.width - player.offsetWidth)); playerY = Math.max(0, Math.min(playerY, canvas.height - player.offsetHeight)); player.style.left = playerX + 'px'; player.style.top = playerY + 'px'; // Kavanoz kontrollerini yap checkCollisions(); // Tüm kavanozlar toplandıysa çıkışı kontrol et if (jarsCollected === 12) { alert('Çıkış kapısını bul ve kaç!'); } });
